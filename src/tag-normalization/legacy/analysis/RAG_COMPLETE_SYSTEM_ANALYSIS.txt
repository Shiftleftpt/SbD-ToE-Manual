PARTE 5: IMPLEMENTAÇÃO PRÁTICA (Pseudocódigo)

```python
from sentence_transformers import SentenceTransformer
from chroma import Chroma  # Vector store
from ollama import Client  # Local LLM
from pathlib import Path

# 1. SETUP (one-time)
embedder = SentenceTransformer('multilingual-MiniLM-L6-v2')
vector_store = Chroma(persist_directory='./manual_index')
llm = Client(base_url='http://localhost:11434')

# 2. INDEX MANUAL
for md_file in Path('docs').rglob('*.md'):
    content = md_file.read_text()
    embedding = embedder.encode(content)
    vector_store.add(
        documents=[content],
        embeddings=[embedding],
        metadatas=[{'path': str(md_file)}]
    )

# 3. QUERY WITH RAG + LLM
def smart_analysis(query, filepath):
    # Retrieve similar docs
    results = vector_store.query(query, n_results=3)
    context = "\\n".join(results['documents'])
    
    # Ask LLM with context
    prompt = f"""
    Based on these similar manual sections:
    {context}
    
    {query}
    """
    
    response = llm.generate(prompt, model='mistral')
    return response

# 4. EXAMPLE USAGE
tags = smart_analysis(
    "Suggest tags for this document",
    filepath="new_doc.md"
)

gaps = smart_analysis(
    "What content is missing compared to similar docs?",
    filepath="03-threat-modeling/01-intro.md"
)
```

Result: Completo sistema RAG local, zero API calls, zero custos, 100% privado
