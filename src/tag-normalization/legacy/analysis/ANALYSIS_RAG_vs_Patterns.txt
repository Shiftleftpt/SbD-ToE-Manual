Here's what Stage 2 (RAG without LLM) would look like:

```python
from sentence_transformers import SentenceTransformer, util
import numpy as np

class RAGTagSuggester:
    def __init__(self, reference_docs):
        self.model = SentenceTransformer('sentence-transformers/multilingual-MiniLM-L6-v2')
        # reference_docs = list of (content, tags) tuples from well-tagged chapters
        
        self.reference_embeddings = []
        self.reference_tags = []
        
        for content, tags in reference_docs:
            emb = self.model.encode(content, convert_to_tensor=True)
            self.reference_embeddings.append(emb)
            self.reference_tags.append(tags)
    
    def suggest_tags(self, document_content, top_k=3, min_similarity=0.5):
        '''Find most similar tagged documents, return their tags'''
        doc_embedding = self.model.encode(document_content, convert_to_tensor=True)
        
        # Find top-K most similar reference documents
        scores = util.pytorch_cos_sim(doc_embedding, self.reference_embeddings)[0]
        top_indices = scores.argsort(reverse=True)[:top_k]
        
        # Collect tags from similar documents
        suggested_tags = {}
        for idx in top_indices:
            if scores[idx] > min_similarity:
                for tag in self.reference_tags[idx]:
                    suggested_tags[tag] = suggested_tags.get(tag, 0) + scores[idx]
        
        # Return tags sorted by combined similarity
        return sorted(suggested_tags.items(), key=lambda x: x[1], reverse=True)

# Usage:
rag = RAGTagSuggester(reference_documents=your_well_tagged_chapters)
tags = rag.suggest_tags(new_document_content)
# Output: [('cra', 2.45), ('regulamentacao', 2.30), ('compliance', 2.10)]
```
